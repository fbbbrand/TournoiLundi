function calculer(input) { const ligne = input.parentElement.parentElement; const colonne3 = parseFloat(input.value || 0); const resultat = (colonne3 / 0.2).toFixed(2); const celluleResultat = ligne.querySelector('td:nth-child(4)'); // Définir la couleur en fonction de la valeur if (resultat < 20) { celluleResultat.style.backgroundColor = '#F53535'; } else if (resultat >= 20 && resultat <= 80) { celluleResultat.style.backgroundColor = '#F5AC35'; } else { celluleResultat.style.backgroundColor = '#23B640'; } celluleResultat.style.color = 'white'; // Définir la couleur du texte en blanc celluleResultat.style.fontWeight = 'bold'; // Mettre le texte en gras celluleResultat.textContent = resultat; }
